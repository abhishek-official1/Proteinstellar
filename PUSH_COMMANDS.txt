# Commands to push to GitHub
# Replace YOUR_REPO_NAME with your actual repository name

# 1. Rename branch to main (GitHub's default)
git branch -M main

# 2. Add remote repository
git remote add origin https://github.com/abhishek-official1/YOUR_REPO_NAME.git

# 3. Push to GitHub
git push -u origin main

# If you need authentication, use a Personal Access Token (PAT):
# - Go to: https://github.com/settings/tokens
# - Generate new token (classic)
# - Select scopes: repo (full control)
# - Copy token and use it as password when prompted

# Verify the push was successful:
git remote -v
